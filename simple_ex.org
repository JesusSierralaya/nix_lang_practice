** Hello world!
#+begin_src nix

** Bind values to names let / in
We bind the value ~10~ to the name ~a~.
#+begin_src nix
let
  a = 10;
in
  a
#+end_src

#+RESULTS:
: 10

** Attribute assigned to a variable
#+begin_src nix
let
  mySet = { y = 3; };
in
mySet.y
#+end_src

#+RESULTS:
: 3
